import java.util.Scanner;

public class New1 {

    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);
        System.out.println("Пожалуйста, введите число:");

        int n = scanner.nextInt();

        int sum = 100;
        for (int i = 1; i <=n; i++){
            sum = sum = i;

            System.out.println(sum);
        }

    }
}
